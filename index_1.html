สินค้า & กำลังการผลิต ตามรอบ (Cycle)

กำลังผลิตต่อวัน = ชิ้นต่อรอบ × รอบต่อคนต่อวัน × จำนวนคน — เพิ่มคน = เพิ่มกำลังผลิต (ใช้ตอนคำนวณ “เร่งส่ง”)
ชื่อสินค้าสต็อกที่มีชิ้น/รอบรอบ/คน/วันจำนวนคน= ผลิต/วัน

วัตถุดิบ (คลัง)

ชื่อวัตถุดิบหน่วยคงเหลือในคลัง

สูตรการผลิต (BOM) — สินค้า 1 ชิ้น ใช้วัตถุดิบอะไรเท่าไร

กรอกปริมาณวัตถุดิบที่ใช้ต่อสินค้า 1 ชิ้น (เว้นว่าง = ไม่ใช้)

ปฏิทินการผลิต

☁️ เชื่อมต่อ Supabase ปิดอยู่

กรอก Project URL และ anon (public) key ของ Supabase — เก็บเฉพาะในเบราว์เซอร์นี้ ไม่ถูกฝังในโค้ด/ไม่ขึ้น GitHub  ·  ⚠️ ใช้ anon key เท่านั้น ห้ามใช้ service_role key เด็ดขาด
📋 SQL สำหรับสร้างตาราง (รันครั้งเดียวใน Supabase → SQL Editor) 
create table if not exists mrp_state (
  id text primary key,
  data jsonb not null default '{}'::jsonb,
  updated_at timestamptz default now()
);
alter table mrp_state enable row level security;
create policy "mrp anon read"  on mrp_state for select using (true);
create policy "mrp anon write" on mrp_state for insert with check (true);
create policy "mrp anon update" on mrp_state for update using (true) with check (true);
⚠️ นโยบายด้านบนเปิดให้ใครก็ตามที่มี URL+anon key อ่าน/เขียนได้ (เหมาะกับใช้ในทีมที่ไว้ใจกัน) หากต้องการจำกัดสิทธิ์ แนะนำเพิ่ม Supabase Auth ภายหลัง — บอกผมได้

ข้อมูล / สำรอง

ข้อมูลถูกบันทึกอัตโนมัติในเบราว์เซอร์นี้ (localStorage) และไม่ถูกส่งออกไปที่ไหน ใช้ปุ่มส่งออก/นำเข้าเพื่อย้ายเครื่องหรือสำรองไฟล์
